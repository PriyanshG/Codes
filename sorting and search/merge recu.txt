#include<bits/stdc++.h>
using namespace std;
int A[10];
void merg(int s,int m,int e)
{
    int p=s,i,q=m+1;
    int B[e-s+1],k=0;
    for(i=s;i<=e;i++)
    {
        if(p>m)
           B[k++]=A[q++];
        else if(q>e)
          B[k++]=A[p++];
         if(A[p]>A[q])
          B[k++]=A[q++];
        else
            B[k++]=A[p++];

    }
    for(p=0;p<k;p++)
    {
        A[s++]=B[p];
    }
}
void merge_sort(int s,int e)
{
    if(s<e)
    {
        int m=(s+e)/2;
        merge_sort(s,m);
        merge_sort(m+1,e);
        merg(s,m,e);
    }
}
int main()
{   int n,i;
    cin>>n;
    for(i=0;i<n;i++)
        cin>>A[i];
    merge_sort(0,n-1);
    for(i=0;i<n;i++)
        cout<<A[i];


}
